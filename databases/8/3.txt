-- находит курсы по которым нет людей с хорошими оценками
-- полезно для нахождения злых преподавателей
select CourseName
from Courses
where
	CourseId not in (
		select Marks.CourseId
		from Marks
		where
			Marks.Mark > 3
	)

-- нужно ли дропать индексы из задания 1 и 2?..
create index MarkToCourseId on Marks using btree (Mark, CourseId);

-- уже был
create index CourseIdIsPK on Courses using hash (CourseId);
